         1 def ben_graham_preprocessing(image, target_size=224):
         2     """
         3     Ben Graham preprocessing for fundus images:
         4     1. Crop borders (remove black edges)
         5     2. Resize to target size
         6     3. Local color averaging (reduce lighting variations)
         7     """
         8     # Convert to grayscale for mask
         9     gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        10     
        11     # Create mask of non-black pixels
        12     _, thresh = cv2.threshold(gray, 10, 255, cv2.THRESH_BINARY)
        13     
        14     # Find contours and get bounding box
        15     contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, 
       cv2.CHAIN_APPR
           OX_SIMPLE)
        16     if contours:
        17         x, y, w, h = cv2.boundingRect(np.vstack(contours))
        18         image = image[y:y+h, x:x+w]
        19     
        20     # Resize
        21     image = cv2.resize(image, (target_size, target_size), 
       interpolation=cv2.
           INTER_AREA)
        22     
        23     # Local color averaging (subtract local mean)
        24     blurred = cv2.GaussianBlur(image, (0, 0), target_size / 30)
        25     image = cv2.addWeighted(image, 4, blurred, -4, 128)
        26     
        27     return image
